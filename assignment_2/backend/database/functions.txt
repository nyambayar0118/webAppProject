/* Fetch functions */
create or replace function getProductsByShopId(
  s_id bigint
)
returns table(product_ids bigint) as $$
begin
 return query
  select id from product
  where shop_id = s_id;
end;
$$ language plpgsql;

create or replace function getProductById(
  p_id bigint
)
return table(name varchar, price bigint, created timestamp, rating float, category varchar, size varchar[], stock bigint, description varchar, shop_id bigint) as $$
begin
 returns query
   select name, price, rating, category_id as category, size, stock, description, shop_id from product
   where id = p_id;
end;
$$ language plpgsql;

/* Verification functions */
create function checkUserName(input_hash varchar)
returns boolean as $$
begin
 return exists(select 1 from login where username = input_hash);
end;
$$ language plpgsql;

create or replace function checkUserPassword(input_hash varchar)
returns boolean as $$
begin
 return exists(select 1 from login where password = input_hash);
end;
$$ language plpgsql;
